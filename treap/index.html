Treap

发布于 2020-11-19 | 分类于 数据结构 | 6分钟 | 1192字数

模板

#include<bits/stdc++.h>
using namespace std;
const int N = 100005;
#define INF (1 << 30)

int n;
struct Treap
{
  int ls[N], rs[N], val[N], rnd[N], sz[N], cnt[N];
  int tot, ans, rt;
  inline void pushup(int x) { sz[x] = sz[ls[x]] + sz[rs[x]] + cnt[x]; }
  void lrotate(int &k)
  {
    int t = rs[k];
    rs[k] = ls[t];
    ls[t] = k;
    sz[t] = sz[k];
    pushup(k);
    k = t;
  }
  void rrotate(int &k)
  {
    int t = ls[k];
    ls[k] = rs[t];
    rs[t] = k;
    sz[t] = sz[k];
    pushup(k);
    k = t;
  }
  void insert(int &k, int x)
  {
    if (!k)
    {
      tot++;
      k = tot;
      sz[k] = 1;
      cnt[k] = 1;
      val[k] = x;
      rnd[k] = rand();
      return;
    }
    sz[k]++;
    if (val[k] == x)
    {
      cnt[k]++;
    }
    else if (val[k] < x)
    {
      insert(rs[k], x);
      if (rnd[rs[k]] < rnd[k])
        lrotate(k);
    }
    else
    {
      insert(ls[k], x);
      if (rnd[ls[k]] < rnd[k])
        rrotate(k);
    }
  }

  void del(int &k, int x)
  {
    if (!k)
      return;
    --sz[k];
    if (val[k] == x)
    {
      if (cnt[k] > 1)
      {
        --cnt[k];
        return;
      }
      if (ls[k] == 0 || rs[k] == 0)
        k = ls[k] + rs[k];
      else if (rnd[ls[k]] < rnd[rs[k]])
      {
        rrotate(k);
        del(rs[k], x);
      }
      else
      {
        lrotate(k);
        del(ls[k], x);
      }
    }
    else if (val[k] < x)
      del(rs[k], x);
    else
      del(ls[k], x);
  }

  int queryrank(int k, int x)
  {
    if (!k)
      return 0;
    if (val[k] == x)
      return sz[ls[k]] + 1;
    else if (x > val[k])
      return sz[ls[k]] + cnt[k] + queryrank(rs[k], x);
    else
      return queryrank(ls[k], x);
  }

  int querynum(int k, int x)
  {
    if (!k)
      return 0;
    if (x <= sz[ls[k]])
      return querynum(ls[k], x);
    else if (x > sz[ls[k]] + cnt[k])
      return querynum(rs[k], x - sz[ls[k]] - cnt[k]);
    else
      return val[k];
  }

  void querypre(int k, int x)
  {
    if (!k)
      return;
    if (val[k] < x)
      ans = k, querypre(rs[k], x);
    else
      querypre(ls[k], x);
  }

  void querysub(int k, int x)
  {
    if (!k)
      return;
    if (val[k] > x)
      ans = k, querysub(ls[k], x);
    else
      querysub(rs[k], x);
  }
} T;

int main()
{
  srand(233);
  scanf("%d", &n);
  int opt, x;
  for (int i = 1; i <= n; i++)
  {
    scanf("%d%d", &opt, &x);
    if (opt == 1)
      T.insert(T.rt, x);
    else if (opt == 2)
      T.del(T.rt, x);
    else if (opt == 3)
    {
      printf("%d\n", T.queryrank(T.rt, x));
    }
    else if (opt == 4)
    {
      printf("%d\n", T.querynum(T.rt, x));
    }
    else if (opt == 5)
    {
      T.ans = 0;
      T.querypre(T.rt, x);
      printf("%d\n", T.val[T.ans]);
    }
    else if (opt == 6)
    {
      T.ans = 0;
      T.querysub(T.rt, x);
      printf("%d\n", T.val[T.ans]);
    }
  }
  return 0;
}

/*
1.插入x数；
2.删除x数(若有多个相同的数，因只删除一个);
3.查询x数的排名(若有多个相同的数，因输出最小的排名);
4.查询排名为x的数；
5.求x的前趋(前趋定义为小于,且最大的数);
6.求x的后继(后继定义为大于,且最小的数)。
*/

基础知识

Treap本质是一颗笛卡尔树，维护的值构成二叉树，修正值构成小根堆，由于修正值是随机生成的，所以treap是趋于随机平衡的。

变量声明及函数

  • ls[N]，rs[N]：i点的左右儿子
  • val[N]，rnd[N]，sz[N]，cnt[N]：权值，修正值，以i为子树的大小，i的数量
  • tot，rt：最大节点号，根节点
  • lrotate(int &k)，rrotate(int &k)，左旋，右旋k节点(旋转后更新k值)
  • insert(int &k, int x)，在以k为root的子树中插入权值x
  • del(int &k, int x)，删除x
  • queryrank(int k, int x)，查询x的排名
  • querynum(int k, int x)，查询排名为x的数
  • querypre(int k, int x)，x直接前继
  • querysub(int k, int x)，x直接后继

旋转


右旋4节点，4节点移至右子

基础操作

  1. 旋转
    e.g.左旋a节点
    • 令b为a的右子
    • a右子指向b左子
    • b左子指向a (a变为b的左子)
    • 更新a值
  2. 插入
    • 当前节点为空节点，建立节点
    • 插入值 >= 当前节点值，插入右子树，回溯时检验修正值，不满足小根堆执行左旋
    • 插入值 < 当前节点值，左子树，回溯检验，不满足，右旋
  3. 删除
    • 若删除节点为链节点或叶节点，删除当前节点，为链节点时返回儿子非空节点
    • 若该节点存在两个非空儿子，通过旋转将该节点旋转为可直接删除节点
    • 旋转逻辑，左子修正值 <= 右子，右旋，访问右子节点
    • 左子 > 右子，左旋，访问左子节点 (朝修正值大的一方旋转)
  4. 找值
    • 同二叉搜索树
  5. 中序访问
    • 左根右
  6. 前驱与后继
    • 贪心，前驱即x前节点中的最大值，val[k] < x 找右子，否者左子
    • 后继，val[k] > x 找左子，否者右子
  7. 查询排名为x的数
    • 找某数前恰好有k - 1个数
    • 递归到空节点，查找失败
    • x <= 左子sz，递归左子
    • x > 左子sz + k节点本身数量，递归右子
    • 否者当前节点即为所求
  8. 查询x的排名
    • 查小于x的节点数量 + 1
    • val[k] == k，返回k节点左子数 + 1
    • 若val[k] < x，ans += k节点sz + k本身sz，递归右子
    • 否者递归左子
